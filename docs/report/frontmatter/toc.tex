% --- TOC entry formatting ---
\titlecontents{chapter}
  [0pt]{\vspace{4pt}\bfseries\normalsize}
  {\contentslabel[\thecontentslabel]{1em}}
  {}
  {\hfill\contentspage}
  []

\titlecontents{section}
  [1.5em]{\vspace{2pt}\small}
  {\contentslabel[\thecontentslabel]{2em}}
  {}
  {\hfill\contentspage}
  []

\titlecontents{subsection}
  [4em]{\vspace{1pt}\small}
  {\contentslabel[\thecontentslabel]{2.5em}}
  {}
  {\hfill\contentspage}
  []

\titlecontents{subsubsection}
  [7em]{\small}
  {\contentslabel[\thecontentslabel]{3em}}
  {}
  {\hfill\contentspage}
  []

% --- Generate TOC ---
\setcounter{page}{5}
\renewcommand{\contentsname}{TABLE OF CONTENTS}
\tableofcontents
\addcontentsline{toc}{chapter}{TABLE OF CONTENTS}
\clearpage

% --- List of Tables ---
\renewcommand{\listtablename}{LIST OF TABLES}
\listoftables
\addcontentsline{toc}{chapter}{LIST OF TABLES}
\clearpage

% --- List of Figures ---
\renewcommand{\listfigurename}{LIST OF FIGURES}
\listoffigures
\addcontentsline{toc}{chapter}{LIST OF FIGURES}
\clearpage
